00017 typedef unsigned char dofObject;  //Bits: 0 0 Tz, Ty, Tz, Z, Y, X. 0 if free, 1 if fixed
00018 enum dofComponent {
00019         X_TRANSLATE=1<<0,
00020         Y_TRANSLATE=1<<1,
00021         Z_TRANSLATE=1<<2,
00022         X_ROTATE=1<<3,
00023         Y_ROTATE=1<<4,
00024         Z_ROTATE=1<<5
00025 };
00026 inline void dofSet(dofObject& obj, dofComponent dof, bool set) {set ? obj|=dof : obj&=~dof;}
00027 inline void dofSetAll(dofObject& obj, bool set) {set ? obj|=0x3F : obj&=~0x3F;}
00028 inline bool dofIsSet(dofObject obj, dofComponent dof){return (dof&obj)?true:false;}
00029 inline bool dofIsAllSet(dofObject obj){return (obj&0x3F)==0x3F;}
00030 inline bool dofIsNoneSet(dofObject obj){return !(obj&0x3F);}
00031 inline dofObject dof(bool tx, bool ty, bool tz, bool rx, bool ry, bool rz) {dofObject ret=0; dofSet(ret, X_TRANSLATE, tx); dofSet(ret, Y_TRANSLATE, ty); dofSet(ret, Z_TRANSLATE, tz); dofSet(ret, X_ROTATE, rx); dofSet(ret, Y_ROTATE, ry); dofSet(ret, Z_ROTATE, rz); return ret;}
