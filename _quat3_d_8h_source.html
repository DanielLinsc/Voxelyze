00161         void FromAngleToPosX(const Vec3D<T>& RotateFrom){ //highly optimized at the expense of readability
00162                 if (Vec3D<T>(0,0,0) == RotateFrom) return; //leave off if it slows down too much!!
00163 
00164                 //Catch and handle small angle:
00165                 T YoverX = RotateFrom.y/RotateFrom.x;
00166                 T ZoverX = RotateFrom.z/RotateFrom.x;
00167                 if (YoverX<SMALL_ANGLE_RAD && YoverX>-SMALL_ANGLE_RAD && ZoverX<SMALL_ANGLE_RAD && ZoverX>-SMALL_ANGLE_RAD){ //Intercept small angle and zero angle
00168                         x=0; y=0.5*ZoverX; z=-0.5*YoverX;
00169                         w = 1+0.5*(-y*y-z*z); //w=sqrt(1-x*x-y*y), small angle sqrt(1+x) ~= 1+x/2 at x near zero.
00170                         return;
00171                 }
00172 
00173                 //more accurate non-small angle:
00174                 Vec3D<> RotFromNorm = RotateFrom;
00175                 RotFromNorm.NormalizeFast(); //Normalize the input...
00176 
00177                 T theta = acos(RotFromNorm.x); //because RotFromNorm is normalized, 1,0,0 is normalized, and A.B = |A||B|cos(theta) = cos(theta)
00178                 if (theta > PI-DISCARD_ANGLE_RAD) {w=0; x=0; y=1; z=0; return;} //180 degree rotation (about y axis, since the vector must be pointing in -x direction

00183                 const T s = sin(a);
